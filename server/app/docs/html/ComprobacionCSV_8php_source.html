142 private function esArchivoCSV($uploadedFile): bool
143 {
144 if (empty($uploadedFile) || !is_uploaded_file($uploadedFile->getStream()->getMetadata('uri'))) {
145 return false;
146 }
147
148
149 $fileExtension = strtolower(pathinfo($uploadedFile->getClientFilename(), PATHINFO_EXTENSION));
150
151 if ($fileExtension !== 'csv') {
152 return false;
153 }
154
155 $fileMimeType = mime_content_type($uploadedFile->getStream()->getMetadata('uri'));
156
157 $allowedMimeTypes = [
158 'text/csv',
159 'text/plain',
160 'application/csv',
161 'text/comma-separated-values',
162 'application/excel',
163 'application/vnd.ms-excel',
164 'application/vnd.msexcel',
165 'text/anytext',
166 'application/octet-stream',
167 'application/txt',
168 ];
169
170 if (!in_array($fileMimeType, $allowedMimeTypes)) {
171 return false;
172 }
173
